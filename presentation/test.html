Touch Events

Exemple1

Récupérer un événement touch et afficher ses coordonnées

monElement.addEventListener('touchstart', afficheTouch, false);
monElement.addEventListener('touchmove', afficheTouch, false);
monElement.addEventListener('touchend', effaceTouch, false);

function afficheTouch(ev) {
	ev.preventDefault();

	monAutreElement.innerHTML = ev.touches.item(0).screenX + "; " + ev.touches.item(0).screenY + "; ";
}

function effaceTouch(ev) {
	document.getElementById('result').innerHTML="";
}
